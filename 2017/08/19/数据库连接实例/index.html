        out.println("数据库连接成功！");

oracle连接数据库

package DB;
import java.sql.*;
public class DBBean {
private String driverStr = "oracle.jdbc.driver.OracleDriver";
private String connStr = "jdbc:oracle:thin:@localhost:1521:orcl";
private String dbusername = "user_20142932";
private String dbpassword = "user_20142932";
private Connection conn = null;
private Statement stmt = null;

public DBBean() {
    try {
        Class.forName(driverStr);
        conn = DriverManager.getConnection(connStr, dbusername, dbpassword);
        stmt = conn.createStatement();
        System.out.println("数据库连接成功！");
        }
    catch (Exception ex) {
        System.out.println("连接数据库失败 \n错误原因： "+ex.getMessage() );
    }
}

public int executeUpdate(String s) {
    int result = 0;
    try {
        System.out.println("执行更新......" );
        result = stmt.executeUpdate(s);
    } 
    catch (Exception ex) 
    {
        System.out.println("执行更新错误\n错误原因：  "+ex.getMessage());
    }
    return result;
}

public ResultSet executeQuery(String s) {
    ResultSet rs = null;
    try {
        System.out.println("执行查询......" );
        rs = stmt.executeQuery(s);
    }
    catch (Exception ex) {
        System.out.println("执行查询错误?\n错误原因： "+ex.getMessage());
    }
    return rs;
}

public void close() {
    try {
        stmt.close();
        conn.close();
    } 
    catch (Exception e) {
        System.out.println("关闭失败?\n错误原因： "+e.getMessage());
    }
    System.out.println("已断开与数据库的连接！" );
}
}

注：除源代码外，还需要相应的jar包
